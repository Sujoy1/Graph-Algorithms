#include <iostream>
#include <bits/stdc++.h>
using namespace std;
class Graph{
    int V;
    list<int> *adj;
    public:
    Graph(int V);
    void addEdge(int v , int w);
    void BFS(int s);
    
};
Graph :: Graph(int V)
    {
        this->V=V;
        adj = new list<int>[V];
    }
void Graph :: addEdge(int v , int w)
    
    {
        //directed graph
        adj[v].push_back(w);
    }
void Graph :: BFS(int s)
    {
     bool  *visited = new bool[V];
     for(int i=0 ; i<V;i++)
     visited[i]=false;
     visited[s]=true;
     queue<int> q;
     q.push(s);
     while(!q.empty())
     {
         int ele = q.front();;
         cout<<ele<<' ';
         q.pop();
         list<int> :: iterator itr;
         for(itr=adj[ele].begin();itr!=adj[ele].end();itr++)
         {
             if(!visited[*itr])
             {
                 visited[*itr]=true;
                 q.push(*itr);
             }
         }
     }
    }    
using namespace std;

int main() {
	Graph g(4);
	g.addEdge(0, 1);
	g.addEdge(0, 2);
    g.addEdge(1, 2);
    g.addEdge(2, 0);
    g.addEdge(2, 3);
    g.addEdge(3, 3);
 
    cout << "Following is Breadth First Traversal "
         << "(starting from vertex 2) \n";
    g.BFS(2);
	return 0;
}
